private static void printFizzBuzz(int n) {
    for (int i = 1; i <= n; i++) {
        if (i % 3 == 0 && i % 5 == 0) {
            System.out.println("FizzBuzz");
        }

            System.out.println("Fizz");

        if (i % 5 == 0) {
            System.out.println("Buzz");
        }
        System.out.println(i);
    }
